<script>
    let currentHeading = 0;
    const headingDisplay = document.getElementById('heading-display');
    const calibrationAlert = document.getElementById('calibration-alert');

    // 取得權限與啟動監聽
    document.getElementById('enable-btn').addEventListener('click', async () => {
        // iOS 需要權限要求，Android 通常不需要但需處理絕對座標
        if (typeof DeviceOrientationEvent.requestPermission === 'function') {
            const permission = await DeviceOrientationEvent.requestPermission();
            if (permission === 'granted') {
                window.addEventListener('deviceorientation', handleOrientation);
            }
        } else {
            // Android 優先使用 deviceorientationabsolute
            if ('ondeviceorientationabsolute' in window) {
                window.addEventListener('deviceorientationabsolute', handleOrientation);
            } else {
                window.addEventListener('deviceorientation', handleOrientation);
            }
        }
        alert("方向監聽已啟動");
        calibrationAlert.style.display = 'block';
    });

    // 平滑化處理 (低通濾波)
    let lastHeading = null;
    const alpha = 0.2; // 數值越小越平滑，但反應會變慢（建議 0.1 ~ 0.3 之間）

    function handleOrientation(event) {
        let heading = 0;

        // 判斷不同瀏覽器的方位數據來源
        if (event.webkitCompassHeading) {
            // iOS 設備
            heading = event.webkitCompassHeading;
        } else if (event.absolute && event.alpha !== null) {
            // Android 設備 (需補償其座標系差異)
            heading = 360 - event.alpha; 
        } else {
            return; // 無法取得準確方位
        }

        // 處理 0/360 度跨越時的旋轉問題
        if (lastHeading !== null) {
            if (heading - lastHeading > 180) {
                lastHeading += 360;
            } else if (lastHeading - heading > 180) {
                lastHeading -= 360;
            }
            // 套用濾波公式：新的值 = 舊值 * (1-α) + 新觀測值 * α
            heading = lastHeading * (1 - alpha) + heading * alpha;
        }

        lastHeading = heading;
        const normalizedHeading = (heading % 360 + 360) % 360;
        
        // 更新 UI
        headingDisplay.innerText = Math.round(normalizedHeading);
        
        // 此處可連動你的 AR 或地圖視角圖標旋轉
        // marker.setRotation(normalizedHeading);
    }
</script>
